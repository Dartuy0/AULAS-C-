#include <stdio.h>

in main(void) {
in matriz[2][3];
int linha, coluna;

for (linha = 0; linha < 2; linha++) {
    for (coluna = 0; coluna < 3: coluna++){
        printf("digite o valor da posicao [%d] [%d]: ", linha, coluna);
        scanf("%d, &matriz[linha][coluna]);
   }
}
printf("\nMatriz digitada:\n");
 for (linha = 0; linha < 2; linha++) {
     for (coluna = 0: coluna < 3; coluna++) {
         printf("%d", matriz[linha][coluna]);
   }
   printf("\n");
}
  return 0;
}
